📖 Google Apps Script Setup Guide

Follow these steps once to connect to Google Sheets:

  1. Go to sheets.google.com → create a new spreadsheet
  2. Name the columns in Row 1:
    Timestamp | Group | Day | Room | S1 Name | S1 ID | E1-S1 Report | E1-S1 Pres | E2-S1 Report | E2-S1 Pres | E3-S1 Report | E3-S1 Pres | S2 Name | S2 ID | E1-S2 Report | E1-S2 Pres | E2-S2 Report | E2-S2 Pres | E3-S2 Report | E3-S2 Pres | S1 Total | S2 Total | Group Avg
  3. Click Extensions → Apps Script
  4. Paste the script code (click copy below) and save
  5. Click Deploy → New Deployment → Web App
  6. Set "Execute as: Me" and "Who has access: Anyone"
  7. Click Deploy → copy the URL → paste it above
  8. Set sheet sharing to View Only for your team, and keep edit access only for yourself

Apps Script Code — copy & paste into your script editor:

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var d = e.parameter;
  
  sheet.appendRow([
    d.timestamp,
    d.groupNumber,
    d.day,
    d.room,
    d.s1Name,
    d.s1Id,
    d.e1s1Report,
    d.e1s1Presentation,
    d.e2s1Report,
    d.e2s1Presentation,
    d.e3s1Report,
    d.e3s1Presentation,
    d.s2Name,
    d.s2Id,
    d.e1s2Report,
    d.e1s2Presentation,
    d.e2s2Report,
    d.e2s2Presentation,
    d.e3s2Report,
    d.e3s2Presentation,
    d.s1Total,
    d.s2Total,
    d.groupAvg
  ]);
  
  return ContentService
    .createTextOutput(JSON.stringify({status: 'success'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status: 'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
